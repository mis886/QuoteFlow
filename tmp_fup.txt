function openFollowupModal(leadId) {
  currentFollowupLeadId = leadId;
  const lead = DB.leads.find(l => l.id === leadId);
  if (!lead) return;

  document.getElementById('followup-modal-title').textContent = `Follow-up #${lead.followupCount + 1} — ${lead.company}`;

  // Set default next followup date to +3 days (max 7)
  const d = new Date(); d.setDate(d.getDate() + 3);
  document.getElementById('fu-date').value = d.toISOString().split('T')[0];

  // Show mini timeline
  const fus = DB.followups[leadId] || [];
  const preview = document.getElementById('followup-timeline-preview');
  if (preview) {
    preview.innerHTML = fus.length === 0 ? '' :
      `<div style="background:var(--gray50);border-radius:8px;padding:10px 14px;font-size:12px;margin-bottom:4px;">
        <div style="font-weight:600;color:var(--gray700);margin-bottom:6px;">Previous follow-ups (${fus.length}/7)</div>
        ${fus.slice(-3).map(f => `<div style="color:var(--gray500);padding:2px 0;border-bottom:1px solid var(--gray200);">${f.date} · ${f.type} · ${f.notes.slice(0,60)}${f.notes.length>60?'…':''}</div>`).join('')}
      </div>`;
  }

  closeDrawer();
  openModal('followup-modal');
}

async function saveFollowup() {
  const leadId = currentFollowupLeadId;
  const lead = DB.leads.find(l => l.id === leadId);
  if (!lead) return;

  const notes = document.getElementById('fu-notes')?.value?.trim();
  const fuDate = document.getElementById('fu-date')?.value;
  if (!notes || !fuDate) { showToast('Notes and next follow-up date are required', 'error'); return; }

  if (lead.followupCount >= 7) {
    showToast('Maximum 7 follow-ups reached. Please close or escalate this lead.', 'error');
    return;
  }

  // Validate max 7 days from last followup
  const lastDate = lead.lastFollowup ? new Date(lead.lastFollowup) : new Date(lead.created);
  const nextDate = new Date(fuDate);
  const diffDays = Math.floor((nextDate - lastDate) / 86400000);
  if (diffDays > 7) {
    showToast('Follow-up must be within 7 days of the last contact.', 'error');
    return;
  }

  const today = new Date().toISOString().split('T')[0];
  const fuType    = document.getElementById('fu-type')?.value || '';
  const fuOutcome = document.getElementById('fu-outcome')?.value || '';
  const stageUpdate = document.getElementById('fu-stage')?.value;
  const userName  = window._currentUser?.name || 'Sales Team';

  const fuRecord = {
    date:              today,
    type:              fuType,
    notes,
    outcome:           fuOutcome,
    by:                userName,
    next_followup_date: fuDate,
  };

  // Save to Supabase — existing followups table
  try {
    const { error } = await _sb.from('followups').insert({
      lead_id:            leadId,
      date:               today,
      type:               fuType,
      notes,
      outcome:            fuOutcome,
      next_followup_date: fuDate,
      stage_update:       stageUpdate || null,
      logged_by:          userName,
    });
    if (error) throw error;
  } catch(e) {
    console.warn('saveFollowup Supabase error:', e.message);
    showToast('Saved locally (Supabase sync failed)', 'warning');
  }

  // Update in-memory state
  if (!DB.followups[leadId]) DB.followups[leadId] = [];
  DB.followups[leadId].push(fuRecord);

  lead.followupCount++;
  lead.lastFollowup = today;
  lead.nextFollowup = fuDate;
  // If stage was changed via followup form, persist it to Supabase
  if (stageUpdate) await updateLeadStage(leadId, stageUpdate);

  closeModal('followup-modal');
  renderPipeline();
  showToast(`Follow-up #${lead.followupCount} logged for ${lead.company}`, 'success');
}

// Load all follow-ups from Supabase (existing followups table) and populate DB.followups
async function loadFollowupsFromSupabase() {
  try {
    const { data, error } = await _sb.from('followups').select('*').order('date', { ascending: true });
    if (error) throw error;
    DB.followups = {};
    (data || []).forEach(r => {
      const key = r.lead_id;
      if (!DB.followups[key]) DB.followups[key] = [];
      DB.followups[key].push({
        id:                r.followup_id,
        date:              r.date,
        type:              r.type,
        notes:             r.notes,
        outcome:           r.outcome,
        by:                r.logged_by,
        next_followup_date: r.next_followup_date,
        stage_update:      r.stage_update,
      });
    });
    // Sync followupCount + next followup date back onto pipeline leads
    DB.leads.forEach(l => {
      const fus = DB.followups[l.id] || [];
      l.followupCount = fus.length;
      if (fus.length > 0) {
        l.lastFollowup = fus[fus.length - 1].date;
        l.nextFollowup = fus[fus.length - 1].next_followup_date || null;
      }
    });
    if (document.getElementById('pipeline-deals-list')) renderPipeline();
  } catch(e) {
    console.warn('loadFollowupsFromSupabase:', e.message);
  }
}
